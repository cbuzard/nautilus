00200       PDJ2(product2_idx) = PDJ2(product2_idx) + tmp_value

00216       PDJ2(reactant2_idx) = PDJ2(reactant2_idx) - tmp_value

00218     endif
00219 
00220     if (reactant2_idx.eq.J) then 
00221       tmp_value = XK(reaction_idx) * Y(reactant1_idx) * Y(reactant3_idx) * H_number_density_squared
00222       PDJ2(product1_idx) = PDJ2(product1_idx) + tmp_value
00223       PDJ2(product2_idx) = PDJ2(product2_idx) + tmp_value

00230 
00231     if (reactant3_idx.eq.J) then

00233       PDJ2(product1_idx) = PDJ2(product1_idx) + tmp_value
00234       PDJ2(product2_idx) = PDJ2(product2_idx) + tmp_value
00235       PDJ2(product3_idx) = PDJ2(product3_idx) + tmp_value

00525   ! ====== Rxn ITYPE 0
00526   ! ITYPE 0: Gas phase reactions with GRAINS =) 
00527   do J=type_id_start(0),type_id_stop(0)
00528     XK(J)=A(J)*(T300**B(J))

00532   ! this is done with the XH species, reaction types 10 and 11

00553     XK(J)=A(J)*(CR_IONISATION_RATE+X_IONISATION_RATE)
00554   enddo
00555

00557   ! Bimolecular gas phase reactions - several possible formula 
00558   W=1

00560   NFIN=0
00561   distmin(:)=9999.d0

00571

00597           if (gas_temperature.GT.Tmax(N)) XK(N)=0.d0
