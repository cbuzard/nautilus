00295   endif
00296

00298   IF (species_name(J).EQ.YE) then
00299     INDEL=J
00300   endif
00301 enddo
00302 
00303 
00304 
00305 ! --- Re-arrange order of elements to match species_composition columns (reactions file)
00306 do J=1,NB_PRIME_ELEMENTS-1
00307   if (species_composition(J,PRIME_ELEMENT_IDX(J)).NE.1) then
00308     do K=J+1,NB_PRIME_ELEMENTS
00309       if (species_composition(J,PRIME_ELEMENT_IDX(K)).EQ.1) then
00310         ISPTEMP=PRIME_ELEMENT_IDX(K)
00311         mass_tmp = elemental_mass(k)
00312         name_tmp = element_name(k)
00313         
00314         elemental_mass(k) = elemental_mass(j)
00315         element_name(k)  = element_name(j)
00316         PRIME_ELEMENT_IDX(K)=PRIME_ELEMENT_IDX(J)
00317         
00318         elemental_mass(k) = mass_tmp
00319         element_name(k) = name_tmp
00320         PRIME_ELEMENT_IDX(J)=ISPTEMP
00321       endif
00322     enddo
00323   endif
00324 enddo
00325 
00326 ! Set species' characteristics==========================================
00327 
00328 ! --- Set reference species
00329 do I=1,nb_species 
00330   ! ------ Calculate elemental_masses
00331   MSUM=0.d0
00332   do K=1,NB_PRIME_ELEMENTS 
00333     MSUM=MSUM+elemental_mass(K)*species_composition(K,I) 
00334   enddo 
00335   AWT(I)=MSUM
00336   if (species_name(I).EQ.YE) AWT(I)=1.D+0/1836.D+0 
00337   if (species_name(I).EQ.YGRAIN .OR. species_name(I).EQ.'GRAIN-      ')&
00338   AWT(I)=4.0*PI*grain_radius*grain_radius*grain_radius*GRAIN_DENSITY/3.0/AMU
00339 enddo
00340

00344 ! initial_dtg_elemental_mass_ratio is changed later into the dust/hydrogen elemental_mass ratio
00345

00348 ! Find ITYPE first and last reactions===================================
00349 do I=0,MAX_NUMBER_REACTION_TYPE-1
00350   type_id_start(I)=0
00351   type_id_stop(I)=0
00352   do J=1,nb_reactions
00353     if ((REACTION_TYPE(J).EQ.I).AND.(type_id_start(I).EQ.0)) type_id_start(I)=J
00354     if (REACTION_TYPE(J).EQ.I) type_id_stop(I)=J
00355   enddo
00356 enddo
00357 
00358 ! Find the index of CO and H2
00359 do i=1,nb_species
00360   if (species_name(i).eq.YH2) INDH2=i
00361   if (species_name(i).eq.YCO) INDCO=i
00362   if (species_name(i).eq.YHE) INDHE=i
00363 enddo
00364 
00365 ! Compute nb_sites_per_grain = number of sites per grain

00384 subroutine integrate_chemical_scheme(delta_t,itol,atol,itask,istate,iopt,mf)
00385 
00386   use global_variables
00387   
00388   implicit none
00389   
00390   ! Inputs
00391   real(double_precision), intent(in) :: delta_t !<[in] time during which we must integrate
00392   integer, intent(in) :: itol
00393   integer, intent(in) :: itask
00394   integer, intent(in) :: iopt
00395   integer, intent(in) :: mf
00396   real(double_precision), intent(in) :: atol
00397 
00398   ! Outputs
00399   integer, intent(out) :: istate
00400  
00401   ! Locals
00402   integer :: i
00403   real(double_precision) :: t !< The local time, starting from 0 to delta_t
00404   real(double_precision), dimension(nb_species) :: satol
00405   real(double_precision), dimension(nb_species) :: temp_abundances
00406 
00407   real(double_precision) :: t_stop_step
00408 
00409   t_stop_step = delta_t
00410   t = 0.d0
00411 
00412   temp_abundances(1:nb_species) = abundances(1:nb_species)
00413 
00414   do while (t.lt.t_stop_step)
00415 
00416     istate = 1
00417 
00418     ! Adaptive absolute tolerance to avoid too high precision on very abundant species,
00419     ! H2 for instance. Helps running a bit faster
00420 
00421     do i=1,nb_species
00422       satol(i)=max(atol,1.d-16*temp_abundances(i))
00423     enddo
00424 
00425     ! Feed IWORK with IA and JA
00426 
00427     call set_work_arrays(Y=temp_abundances)
00428     
00429 
00430     call dlsodes(get_temporal_derivatives,nb_species,temp_abundances,t,t_stop_step,itol,RELATIVE_TOLERANCE,&
00431     satol,itask,istate,iopt,rwork,lrw,iwork,liw,get_jacobian,mf)       
00432 
00433     ! Whenever the solver fails converging, print the reason.
00434     ! cf odpkdmain.f for translation
00435     if (istate.ne.2) then
00436       write(*,*)  'ISTATE = ', ISTATE
00437     endif
00438 
00439     call check_conservation(temp_abundances)
00440 
00441   enddo
00442 
00443   abundances(1:nb_species) = temp_abundances(1:nb_species)
00444 
00445   return 
00446   end subroutine integrate_chemical_scheme

00459   subroutine check_conservation(temp_abundances)
00460   use global_variables
00461   
00462   implicit none
00463   
00464   ! Inputs/outputs
00465   real(double_precision), intent(inout), dimension(nb_species) :: temp_abundances !< [in,out] The list of abundances for all species. 
00466 !! Can be overwritten if CONSERVATION_TYPE is on, and force conservation.
00467   
00468   ! Locals
00469   real(double_precision), dimension(NB_PRIME_ELEMENTS) :: elemental_abundance
00470   real(double_precision) :: CHASUM
00471 
00472   integer :: i, k
00473 
00474   ! Prevent too low abundances
00475   do i=1,nb_species
00476     if (temp_abundances(i).le.1.d-99) then
00477       temp_abundances(i) = 1.d-99
00478     endif
00479   enddo
00480 
00481   ! --- Conserve electrons
00482   CHASUM=0.d0
00483   do I=1,nb_species
00484     if (I.NE.INDEL) CHASUM=CHASUM+SPECIES_CHARGE(I)*temp_abundances(I)
00485   enddo
00486   if (CHASUM.LE.0.d0) CHASUM=MINIMUM_INITIAL_ABUNDANCE
00487   temp_abundances(INDEL)=CHASUM
00488 
00489   ! --- Conserve other elements if selected
00490   if (CONSERVATION_TYPE.GT.0) then
00491     do K=1,CONSERVATION_TYPE
00492       elemental_abundance(K)=0.d0
00493     enddo
00494     do I=1,nb_species
00495       do K=1,CONSERVATION_TYPE
00496         if (I.NE.PRIME_ELEMENT_IDX(K)) elemental_abundance(K)=elemental_abundance(K)+species_composition(K,I)*temp_abundances(I)
00497       enddo
00498     enddo
00499     do K=1,CONSERVATION_TYPE
00500       temp_abundances(PRIME_ELEMENT_IDX(K))=INITIAL_ELEMENTAL_ABUNDANCE(K)-elemental_abundance(K)
00501       if (temp_abundances(PRIME_ELEMENT_IDX(K)).LE.0.d0) temp_abundances(PRIME_ELEMENT_IDX(K))=MINIMUM_INITIAL_ABUNDANCE
00502     enddo
00503   endif
00504 
00505   ! Check for conservation
00506   call get_elemental_abundance(all_abundances=temp_abundances, el_abundances=elemental_abundance)
00507   
00508   call write_elemental_abundances(filename='elemental_abundances.tmp', el_abundances=elemental_abundance)
00509 
00510   do k=1,NB_PRIME_ELEMENTS
00511     if (abs(INITIAL_ELEMENTAL_ABUNDANCE(K)-elemental_abundance(K))/INITIAL_ELEMENTAL_ABUNDANCE(K).ge.0.01d0) then

00514                            INITIAL_ELEMENTAL_ABUNDANCE(K)

00523         write(Error_unit,*) 'He is too depleted on the grains !!!!'
00524       endif
00525     endif       
00526   enddo
00527 
00528   ! VW fev 2012 add a test for the helium and H2 abundance in the gas phase
00529   ! prevent excessive depletion
00530 
00531 
00532   return
00533   end subroutine check_conservation

00581     use global_variables
00582     
00583     implicit none
00584 
00585     ! Locals
00586     real(double_precision), dimension(nb_species) :: REA1,REA2,REA3,REA4

00588     real(double_precision), dimension(nb_species) :: SMASS

00591     integer :: NGS,NEA,NPATH,NEVAP,BADFLAG,ATOMS
00592     character(len=11), dimension(5,nb_reactions) :: GSREAD

00599     character(len=80) :: line

00606     ! Set accretion rate====================================================
00607 
00608     ! COND is used to calculate R_acc = (sigma_d) * <v_i> * n_i * n_d
00609     ! Uses 'Mean' Maxwellian speed, rather than RMS (is this justifiable?)
00610 
00611     COND=PI*grain_radius*grain_radius*SQRT(8.0d0*K_B/PI/AMU)
00612 
00613     ! --- Evaluate sticking coeff and accretion rate factor for each species
00614     STICK=0.d0
00615     do I=1,nb_species
00616       if (SPECIES_CHARGE(I).EQ.0) then
00617         STICK=sticking_coeff_neutral
00618       endif 
00619       if (SPECIES_CHARGE(I).GT.0) then 
00620         STICK=sticking_coeff_positive 
00621       endif
00622       if (SPECIES_CHARGE(I).LT.0) then 
00623         STICK=sticking_coeff_negative 
00624       endif
00625       !         if (species_name(I).EQ.YH2)      STICK=0.D+0 
00626       !         if (species_name(I).EQ.YHE)      STICK=0.D+0 
00627       !         if (species_name(I).EQ.YH)       STICK=0.D+0 
00628       if (species_name(I).EQ.YHEP)     STICK=0.D+0 
00629       if (species_name(I).EQ.'e-         ')      STICK=0.D+0
